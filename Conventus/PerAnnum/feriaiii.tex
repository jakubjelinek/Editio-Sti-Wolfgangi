\ifx\precestotum\undefined
\pars{Preces.}

\sineinitiali{}{temporalia/tonusprecumnovum.gtex}

\ifx\preces\undefined
\ifx\lauda\undefined
\else
\noindent Vocatiónis cæléstis partícipes facti,~\gredagger{} fratres caríssimi,~\grestar{} Iesum, pontíficem confessiónis nostræ, benedicámus clamántes:

\Rbardot{} Dómine Deus noster et salvátor noster.

\noindent O Rex omnípotens,~\gredagger{} qui, per baptísmum, regále nobis sacerdótium contulísti,~\grestar{} fac ut laudis tibi semper sacrifícium offerámus.

\Rbardot{} Dómine Deus noster et salvátor noster.

\noindent Da nobis mandáta tua serváre,~\grestar{} ut per Sanctum Spíritum in te maneámus et tu in nobis.

\Rbardot{} Dómine Deus noster et salvátor noster.

\noindent Sapiéntiam tuam da nobis ætérnam,~\grestar{} ut nobíscum sit hódie et nobíscum operétur.

\Rbardot{} Dómine Deus noster et salvátor noster.

\noindent Concéde nobis hódie néminem umquam contristáre,~\grestar{} omnes autem, qui nobíscum sunt, lætificáre.

\Rbardot{} Dómine Deus noster et salvátor noster.
\fi
\ifx\laudb\undefined
\else
\noindent Salvatóri nostro benedicámus,~\gredagger{} qui sua resurrectióne mundum clarificávit,~\grestar{} et humíliter invocémus eum dicéntes:

\Rbardot{} Salva nos, Dómine, in sémita tua.

\noindent Resurrectiónem tuam, Dómine Iesu,~\gredagger{} oratióne cólimus matutína,~\grestar{} spes glóriæ tuæ diem nostrum illúminet.

\Rbardot{} Salva nos, Dómine, in sémita tua.

\noindent Súscipe, Dómine, vota et propósita nostra,~\grestar{} tamquam diéi nostri primítias.

\Rbardot{} Salva nos, Dómine, in sémita tua.

\noindent Tríbue in dilectióne tua nos hódie profícere,~\grestar{} ut ómnia in nostrum omniúmque bonum cooperéntur.

\Rbardot{} Salva nos, Dómine, in sémita tua.

\noindent Da, Dómine, sic lucére lucem nostram coram homínibus,~\grestar{} ut vídeant ópera nostra bona et Patrem gloríficent.

\Rbardot{} Salva nos, Dómine, in sémita tua.
\fi
\ifx\laudc\undefined
\else
\noindent Christum, qui sibi pópulum novum acquisívit in sánguine suo,~\grestar{} adorémus et rogémus supplíciter:

\Rbardot{} Meménto pópuli tui, Dómine.

\noindent Rex et Redémptor noster, audi Ecclésiam tuam, huius diéi inítio te laudántem,~\grestar{} doce eam sine intermissióne glorificáre maiestátem tuam.

\Rbardot{} Meménto pópuli tui, Dómine.

\noindent Spes et fírmitas nostra, in te confídimus,~\grestar{} neque umquam confundémur.

\Rbardot{} Meménto pópuli tui, Dómine.

\noindent Infirmitátem nostram réspice et in adiutórium nostrum festína,~\grestar{} quia sine te nihil póssumus fácere.

\Rbardot{} Meménto pópuli tui, Dómine.

\noindent Páuperum et derelictórum meménto, ne dies novus sit eis pondus,~\grestar{} sed solámen et gáudium.

\Rbardot{} Meménto pópuli tui, Dómine.
\fi
\ifx\laudd\undefined
\else
\noindent Deus, lætítiam nobis concédens hoc matutíno témpore eum laudándi,~\gredagger{} spem róborat nostram.~\grestar{} Ideo fidénter ipsi supplicémus:

\Rbardot{} Ad glóriam nóminis tui, audi nos, Dómine.

\noindent Grátias ágimus tibi,~\gredagger{} Deus et Pater Salvatóris nostri Iesu,~\grestar{} pro cognitióne et immortalitáte, quas dedísti nobis per ipsum.

\Rbardot{} Ad glóriam nóminis tui, audi nos, Dómine.

\noindent Concéde humilitátem córdibus nostris,~\grestar{} ut subiécti simus ínvicem in timóre Christi.

\Rbardot{} Ad glóriam nóminis tui, audi nos, Dómine.

\noindent Spíritum tuum nobis fámulis tuis infúnde,~\grestar{} ut diléctio nostra fratérna sine simulatióne fiat.

\Rbardot{} Ad glóriam nóminis tui, audi nos, Dómine.

\noindent Qui hómini labórem præcepísti in mundi dominatiónem,~\grestar{} concéde, ut labor noster te célebret fratrésque nostros sanctíficet.

\Rbardot{} Ad glóriam nóminis tui, audi nos, Dómine.
\fi
